var categoriesOrder = Object.keys(generatorConfig);
var productsData = {};

categoriesOrder.forEach(function(cat) {
    productsData[cat] = [];
    var config = generatorConfig[cat];
    for (var i = 1; i <= 100; i++) {
        var pref = config.prefixes[i % config.prefixes.length];
        var noun = config.nouns[(i + 2) % config.nouns.length];
        productsData[cat].push(pref + " " + noun + " #" + i);
    }
});

function buildMainGrid() {
    gridWrap.innerHTML = '';
    categoriesOrder.forEach(function(cat, idx) {
        var card = document.createElement('div');
        card.className = 'item-card';
        card.style.setProperty('--idx', idx);
        card.setAttribute('onclick', "handleCardClick(this, '" + cat + "')");
        card.setAttribute('data-category', cat.toLowerCase());

        var img = document.createElement('div');
        img.className = 'item-img';
        img.style.backgroundImage = "url('https://placehold.co/600x400/1a1a2e/FFF?text=" + encodeURIComponent(cat) + "')";

        var info = document.createElement('div');
        info.className = 'item-info';

        var title = document.createElement('div');
        title.className = 'item-title';
        title.textContent = cat;

        var meta = document.createElement('div');
        meta.className = 'item-meta';
        meta.textContent = "100 объявлений";

        info.appendChild(title);
        info.appendChild(meta);
        card.appendChild(img);
        card.appendChild(info);
        gridWrap.appendChild(card);
    });
}
buildMainGrid();

function buildSidebar() {
    sidebarTarget.innerHTML = '';
    categoriesOrder.forEach(function(cat, i){

        item.className = 'sidebar-item';
        item.style.setProperty('--idx', i);

        item.appendChild(title);

    });

/* Логика Поиска по Товарам (после подтверждения Enter) */
var searchTimeout = null;
searchInput.addEventListener('keydown', function(e) {
    if (e.key === 'Enter') {
        e.preventDefault();
        var value = searchInput.value.toLowerCase().trim();
        
        if (!value) {
            // Если поиск пустой - показываем все карточки
            var cards = gridWrap.querySelectorAll('.item-card');
            cards.forEach(function(card) {
                card.classList.remove('search-hidden');
            });
            return;
        }
        
        // Скрываем все карточки сначала
        var cards = gridWrap.querySelectorAll('.item-card');
        cards.forEach(function(card) {
            card.classList.add('search-hidden');
        });
        
        // Ищем совпадения во всех товарах всех категорий
        categoriesOrder.forEach(function(cat) {
            var items = productsData[cat] || [];
            items.forEach(function(pName) {
                if (pName.toLowerCase().indexOf(value) !== -1) {
                    // Нашли товар - показываем карточку категории
                    var catCard = gridWrap.querySelector('.item-card[data-category="' + cat.toLowerCase() + '"]');
                    if (catCard) {
                        catCard.classList.remove('search-hidden');
                    }
                }
            });
        });
    }
});

// Очистка поиска при клике на крестик или ручном очищении
searchInput.addEventListener('input', function(e) {
    if (e.target.value.trim() === '') {
        var cards = gridWrap.querySelectorAll('.item-card');
        cards.forEach(function(card) {
            card.classList.remove('search-hidden');
        });
    }
});

function initScrollBanner(holder, banner) {
    holder.addEventListener('scroll', function() {
        if (holder.scrollTop > 150) { 
            banner.classList.add('visible');
        } else {
            banner.classList.remove('visible');
        }
    });
    banner.addEventListener('click', function() {
        holder.scrollTo({ top: 0, behavior: 'smooth' });
    });
}

window.handleCardClick = function(card, categoryName) {
    if (phase !== 'idle') return;
    phase = 'animating';

    mainScrollTopBtn.classList.remove('visible');
    currentOriginRect = card.getBoundingClientRect();
    var headerRect = headerBlock.getBoundingClientRect();

    var ghost = document.createElement('div');
    ghost.className = 'card-ghost';
    ghost.style.left = currentOriginRect.left + 'px';
    ghost.style.top = currentOriginRect.top + 'px';
    ghost.style.width = currentOriginRect.width + 'px';
    ghost.style.height = currentOriginRect.height + 'px';
    document.body.appendChild(ghost);

    var holder = document.createElement('div');
    holder.className = 'products-holder';

    var topBanner = document.createElement('button');
    topBanner.type = 'button';
    topBanner.className = 'scroll-top-banner';
    topBanner.innerHTML = 'Наверх ▲';
    holder.appendChild(topBanner);

    var grid = document.createElement('div');
    grid.className = 'products-grid';
    
    var items = productsData[categoryName] || [];
    var baseImgId = generatorConfig[categoryName].imgId;

    items.forEach(function(pName, idx){
        var pCard = document.createElement('div');
        pCard.className = 'product-card';
        pCard.style.setProperty('--p-delay', (Math.min(idx, 15) * 0.02) + 's');
        
        var img = document.createElement('div');
        img.className = 'prod-img';
        
        img.style.backgroundImage = "url('https://placehold.co/300x250/1a1a2e/FFF?text=" + encodeURIComponent(pName) + "')";
        
        var info = document.createElement('div');
        info.className = 'prod-info';
        
        var title = document.createElement('div');
        title.className = 'prod-title';
        title.textContent = pName;
        
        var meta = document.createElement('div');
        meta.className = 'prod-meta';
        meta.textContent = (1500 + (idx * 250)) + " ₽";
        
        info.appendChild(title);
        info.appendChild(meta);
        pCard.appendChild(img);
        pCard.appendChild(info);
        grid.appendChild(pCard);
    });
    
    holder.appendChild(grid);
    ghost.appendChild(holder);

    initScrollBanner(holder, topBanner);
    gridWrap.classList.add('hidden-grid');

    var targetLeft = 54 + 30; 
    var targetTop = headerRect.height + 40;
    var targetWidth = window.innerWidth - targetLeft - 315 - 45; 
    var targetHeight = window.innerHeight - targetTop - 40;

    ghost.getBoundingClientRect();
    ghost.classList.add('fade-in');
    ghost.style.left = targetLeft + 'px';
    ghost.style.top = targetTop + 'px';
    ghost.style.width = targetWidth + 'px';
    ghost.style.height = targetHeight + 'px';

    setTimeout(function(){
        backBtn.classList.add('visible');
        sidebarTarget.classList.add('active');
        highlightSidebar(categoryName);
        phase = 'modal';
    }, 500);
};

window.slideBack = function() {
    if (phase !== 'modal') return;
    phase = 'animating';

    var ghost = document.querySelector('.card-ghost');
    backBtn.classList.remove('visible');
    sidebarTarget.classList.add('leaving');
    sidebarTarget.classList.remove('active');

    var activeBanner = document.querySelector('.scroll-top-banner');
    if(activeBanner) activeBanner.classList.remove('visible');

    setTimeout(function(){
        sidebarTarget.classList.remove('leaving');
        var items = sidebarTarget.querySelectorAll('.sidebar-item');
        items.forEach(function(el){ el.classList.remove('active-category'); });
    }, 400);

    if (ghost && currentOriginRect) {
        ghost.classList.remove('fade-in');
        ghost.style.left = currentOriginRect.left + 'px';
        ghost.style.top = currentOriginRect.top + 'px';
        ghost.style.width = currentOriginRect.width + 'px';
        ghost.style.height = currentOriginRect.height + 'px';
    }

    setTimeout(function(){ gridWrap.classList.remove('hidden-grid'); }, 150);

    setTimeout(function(){
        if (ghost) ghost.remove();
        phase = 'idle';
        if (gridWrap.scrollTop > 150) { mainScrollTopBtn.classList.add('visible'); }
    }, 500);
};

function highlightSidebar(categoryName) {
    var items = sidebarTarget.querySelectorAll('.sidebar-item');
    items.forEach(function(el){
        var t = el.querySelector('.sb-title').textContent;
        if (t === categoryName) { el.classList.add('active-category'); } 
        else { el.classList.remove('active-category'); }
    });
}
